CHANGE 3: Password Hashing (authenticatedUsers.ts, Lines 44-63)

async function handleSignup (req: Request, res: Response, next: NextFunction) { try { const email = req.body.email if (!validator.isEmail(email)) { return res.status(400).send('Invalid email') } const hashedPassword = await bcrypt.hash(req.body.password, 10) const user = { email: req.body.email, password: hashedPassword } const token = jwt.sign({ id: user.email }, 'secret-key') res.json({ token, user }) } catch (error) { next(error) } }
Purpose: Securely hash passwords using bcrypt (10 salt rounds)
Benefit: Even if database is breached, passwords cannot be recovered. Industry standard protection.
